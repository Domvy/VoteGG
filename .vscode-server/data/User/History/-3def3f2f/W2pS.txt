server {
    listen 443 ssl;
    server_name 3.38.211.217; # 실제 IP 주소를 사용하거나 도메인 이름으로 변경

    ssl_certificate /home/ubuntu/Project-Agora/server/cert.pem;
    ssl_certificate_key /home/ubuntu/Project-Agora/server/private.key;

    # Socket.io
    location /socket.io/ {
        proxy_pass http://127.0.0.1:3000/socket.io/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Timeouts
        proxy_read_timeout 600s; # 긴 타임아웃 설정
        proxy_send_timeout 600s;
        proxy_buffering off;
        proxy_cache off;
        }

    # React 애플리케이션 프록시 설정
    location / {
        proxy_pass http://localhost:3001/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
